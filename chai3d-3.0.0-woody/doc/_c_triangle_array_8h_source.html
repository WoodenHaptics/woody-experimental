364  cVector3d getTexCoordAtPosition(const unsigned int a_triangleIndex,
365  const cVector3d& a_localPos)
366  {
367  // get vertices of contact triangles
368  cVector3d vertex0 = m_vertices->getLocalPos(getVertexIndex0(a_triangleIndex));
369  cVector3d vertex1 = m_vertices->getLocalPos(getVertexIndex1(a_triangleIndex));
370  cVector3d vertex2 = m_vertices->getLocalPos(getVertexIndex2(a_triangleIndex));
371 
372  // project desired point on triangle
373  double a0 = 0;
374  double a1 = 0;
375  cProjectPointOnPlane(a_localPos, vertex0, vertex1, vertex2, a0, a1);
376  cVector3d vertex = cAdd(vertex0, cMul(a0, cSub(vertex1, vertex0)), cMul(a1, cSub(vertex2, vertex0)));
377 
378  // compute area of triangle
379  double area = cTriangleArea(vertex0, vertex1, vertex2);
380 
381  // compute areas of three sub-triangles formed by the three vertices of the triangle and the contact point.
382  double area0 = cTriangleArea(vertex, vertex1, vertex2);
383  double area1 = cTriangleArea(vertex, vertex0, vertex2);
384  double area2 = cTriangleArea(vertex, vertex0, vertex1);
385 
386  // compute weights based on position of contact point
387  double c0, c1, c2;
388  if (area > 0.0)
389  {
390  c0 = area0/area;
391  c1 = area1/area;
392  c2 = area2/area;
393  }
394  else
395  {
396  c0 = c1 = c2 = (1.0/3.0);
397  }
398 
399  cVector3d texCoord;
400 
401  if (m_vertices->getUseTexCoordData())
402  {
403  // retrieve the texture coordinate for each triangle vertex
404  cVector3d texCoord0 = m_vertices->getTexCoord(getVertexIndex0(a_triangleIndex));
405  cVector3d texCoord1 = m_vertices->getTexCoord(getVertexIndex1(a_triangleIndex));
406  cVector3d texCoord2 = m_vertices->getTexCoord(getVertexIndex2(a_triangleIndex));
407 
408  // compute the exact texture coordinate at the contact point
409  texCoord = cAdd( cMul(c0, texCoord0), cMul(c1, texCoord1), cMul(c2, texCoord2));
410  }
411  else
412  {
413  texCoord.set(0.0, 0.0, 0.0);
414  }
415 
416  // return result
417  return (texCoord);
418  }

426  inline void computeBTN()
427  {
428  unsigned int numTriangles = getNumTriangles();
429  for (unsigned i=0; i<numTriangles; i++)
430  {
431  cVector3d T,B,N;
432 
433  unsigned int index0 = getVertexIndex0(i);
434  unsigned int index1 = getVertexIndex1(i);
435  unsigned int index2 = getVertexIndex2(i);
436 
437  // Calculate the vectors from the current vertex to the two other vertices in the triangle
438  cVector3d v1v0 = m_vertices->getLocalPos(index1) - m_vertices->getLocalPos(index0);
439  cVector3d v2v0 = m_vertices->getLocalPos(index2) - m_vertices->getLocalPos(index0);
440 
441  cVector3d tex0 = m_vertices->getTexCoord(index0);
442  cVector3d tex1 = m_vertices->getTexCoord(index1);
443  cVector3d tex2 = m_vertices->getTexCoord(index2);
444 
445  // Calculate c1c0_T and c1c0_B
446  double c1c0_T = tex1.x() - tex0.x();
447  double c1c0_B = tex1.y() - tex0.y();
448 
449  // Calculate c2c0_T and c2c0_B
450  double c2c0_T = tex2.x() - tex0.x();
451  double c2c0_B = tex2.y() - tex0.y();
452 
453  double fDenominator = c1c0_T * c2c0_B - c2c0_T * c1c0_B;
454  if (cAbs(fDenominator) < C_TINY)
455  {
456  // We won't risk a divide by zero, so set the tangent matrix to the identity matrix
457  T = cVector3d(1.0, 0.0, 0.0);
458  B = cVector3d(0.0, 1.0, 0.0);
459  }
460  else
461  {
462  // Calculate the reciprocal value once and for all (to achieve speed)
463  double fScale1 = 1.0f / fDenominator;
464 
465  // T and B are calculated just as the equation in the article states
466  T = cVector3d((c2c0_B * v1v0.x() - c1c0_B * v2v0.x()) * fScale1,
467  (c2c0_B * v1v0.y() - c1c0_B * v2v0.y()) * fScale1,
468  (c2c0_B * v1v0.z() - c1c0_B * v2v0.z()) * fScale1);
469 
470  B = cVector3d((-c2c0_T * v1v0.x() + c1c0_T * v2v0.x()) * fScale1,
471  (-c2c0_T * v1v0.y() + c1c0_T * v2v0.y()) * fScale1,
472  (-c2c0_T * v1v0.z() + c1c0_T * v2v0.z()) * fScale1);
473 
474  // Calculate the reciprocal value once and for all (to achieve speed)
475  /*
476  double fScale2 = 1.0f / ((T.x() * B.y() * N.z() - T.z() * B.y() * N.x()) +
477  (B.x() * N.y() * T.z() - B.z() * N.y() * T.x()) +
478  (N.x() * T.y() * B.z() - N.z() * T.y() * B.x()));
479  */
480 
481  // flipped variables (to be checked!)
482  m_vertices->m_tangent[i] = cNormalize(B);
483  m_vertices->m_bitangent[i] = cNormalize(T);
484  }
485  }
486  }

553  inline double computeArea(const unsigned int a_triangleIndex)
554  {
555  // A = 0.5 * | u x v |
556  cVector3d u = m_vertices->getLocalPos(getVertexIndex1(a_triangleIndex)) - m_vertices->getLocalPos(getVertexIndex0(a_triangleIndex));
557  cVector3d v = m_vertices->getLocalPos(getVertexIndex2(a_triangleIndex)) - m_vertices->getLocalPos(getVertexIndex0(a_triangleIndex));
558 
559  return (0.5 * (cCross(u,v).length()));
560  }

574  cVector3d computeNormal(const unsigned int a_triangleIndex,
575  const bool a_applyNormalToVertices)
576  {
577  // get pointer to vertices
578  int vertex0 = getVertexIndex0(a_triangleIndex);
579  int vertex1 = getVertexIndex1(a_triangleIndex);
580  int vertex2 = getVertexIndex2(a_triangleIndex);
581 
582  // retrieve vertex positions
583  cVector3d pos0 = m_vertices->getLocalPos(vertex0);
584  cVector3d pos1 = m_vertices->getLocalPos(vertex1);
585  cVector3d pos2 = m_vertices->getLocalPos(vertex2);
586 
587  // compute normal vector
588  cVector3d normal, v01, v02;
589  pos1.subr(pos0, v01);
590  pos2.subr(pos0, v02);
591  v01.crossr(v02, normal);
592  double length = normal.length();
593  if (length > 0.0)
594  {
595  normal.div(length);
596  if (a_applyNormalToVertices)
597  {
598  m_vertices->setNormal(vertex0, normal);
599  m_vertices->setNormal(vertex1, normal);
600  m_vertices->setNormal(vertex2, normal);
601  }
602  return (normal);
603  }
604 
605  normal.zero();
606  return (normal);
607  }
