750  inline void render()
751  {
752 #ifdef C_USE_OPENGL
753  // create buffers first time
754  if (m_positionBuffer == -1)
755  {
756  glGenBuffers(1, &m_positionBuffer);
757  glBindBuffer(GL_ARRAY_BUFFER, m_positionBuffer);
758  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_localPos[0]), GL_STATIC_DRAW);
759  glEnableVertexAttribArray(C_VB_POSITION);
760  glVertexAttribPointer(C_VB_POSITION, 3, GL_DOUBLE, GL_FALSE, 0, 0);
761  glBindBuffer(GL_ARRAY_BUFFER, 0);
762  }
763 
764  if ((m_normalBuffer == -1) && (m_useNormalData))
765  {
766  glGenBuffers(1, &m_normalBuffer);
767  glBindBuffer(GL_ARRAY_BUFFER, m_normalBuffer);
768  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_normal[0]), GL_STATIC_DRAW);
769  glEnableVertexAttribArray(C_VB_NORMAL);
770  glVertexAttribPointer(C_VB_NORMAL, 3, GL_DOUBLE, GL_FALSE, 0, 0);
771  glBindBuffer(GL_ARRAY_BUFFER, 0);
772  }
773 
774  if ((m_texCoordBuffer == -1) && (m_useTexCoordData))
775  {
776  glGenBuffers(1, &m_texCoordBuffer);
777  glBindBuffer(GL_ARRAY_BUFFER, m_texCoordBuffer);
778  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_texCoord[0]), GL_STATIC_DRAW);
779  glEnableVertexAttribArray(C_VB_TEXCOORD);
780  glVertexAttribPointer(C_VB_TEXCOORD, 3, GL_DOUBLE, GL_FALSE, 0, 0);
781  glBindBuffer(GL_ARRAY_BUFFER, 0);
782  }
783 
784  if ((m_colorBuffer == -1) && (m_useColorData))
785  {
786  glGenBuffers(1, &m_colorBuffer);
787  glBindBuffer(GL_ARRAY_BUFFER, m_colorBuffer);
788  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cColorf), &(m_color[0]), GL_STATIC_DRAW);
789  glEnableVertexAttribArray(C_VB_COLOR);
790  glVertexAttribPointer(C_VB_COLOR, 4, GL_FLOAT, GL_FALSE, sizeof(cColorf), 0);
791  glBindBuffer(GL_ARRAY_BUFFER, 0);
792  }
793 
794  if ((m_tangentBuffer == -1) && (m_useTangentData))
795  {
796  glGenBuffers(1, &m_tangentBuffer);
797  glBindBuffer(GL_ARRAY_BUFFER, m_tangentBuffer);
798  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_tangent[0]), GL_STATIC_DRAW);
799  glEnableVertexAttribArray(C_VB_TANGENT);
800  glVertexAttribPointer(C_VB_TANGENT, 3, GL_DOUBLE, GL_FALSE, 0, 0);
801  glBindBuffer(GL_ARRAY_BUFFER, 0);
802  }
803 
804  if ((m_bitangentBuffer == -1) && (m_useBitangentData))
805  {
806  glGenBuffers(1, &m_bitangentBuffer);
807  glBindBuffer(GL_ARRAY_BUFFER, m_bitangentBuffer);
808  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_bitangent[0]), GL_STATIC_DRAW);
809  glEnableVertexAttribArray(C_VB_BITANGENT);
810  glVertexAttribPointer(C_VB_BITANGENT, 3, GL_DOUBLE, GL_FALSE, 0, 0);
811  glBindBuffer(GL_ARRAY_BUFFER, 0);
812  }
813 
814  // update buffers if needed
815  if (m_flagPositionData)
816  {
817  glBindBuffer(GL_ARRAY_BUFFER, m_positionBuffer);
818  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_localPos[0]));
819  m_flagPositionData = false;
820  }
821  if (m_flagNormalData)
822  {
823  glBindBuffer(GL_ARRAY_BUFFER, m_normalBuffer);
824  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_normal[0]));
825  m_flagNormalData = false;
826  }
827  if (m_flagTexCoordData)
828  {
829  glBindBuffer(GL_ARRAY_BUFFER, m_texCoordBuffer);
830  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_texCoord[0]));
831  m_flagTexCoordData = false;
832  }
833  if (m_flagColorData)
834  {
835  glBindBuffer(GL_ARRAY_BUFFER, m_colorBuffer);
836  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cColorf), &(m_color[0]));
837  m_flagColorData = false;
838  }
839  if (m_flagTangentData)
840  {
841  glBindBuffer(GL_ARRAY_BUFFER, m_tangentBuffer);
842  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_tangent[0]));

848  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_bitangent[0]));

868  inline void allocateData(const int a_numberOfVertices,
869  const bool a_useNormalData,
870  const bool a_useTexCoordData,
871  const bool a_useColorData,
872  const bool a_useTangentData,
873  const bool a_useBitangentData,
874  const bool a_useUserData)
875  {

881  m_localPos.resize(m_numVertices, pos);
882  m_globalPos.resize(m_numVertices, pos);
883 
884  // update normal data allocation

886  if (m_useNormalData)

890  {

892  }
893  else
894  {
895  m_normal.push_back(normal);
896  }
897 
898  }

901  m_normal.clear();
902  }

910  {
911  m_texCoord.resize(m_numVertices, texCoord);

913  else
914  {
915  m_texCoord.push_back(texCoord);
916  }

920  m_texCoord.clear();
921  }
922 
923  // update color data allocation
924  m_useColorData = a_useColorData;
925  if (m_useColorData)
926  {
927  cColorf color(0.0, 0.0, 0.0, 1.0);
928  if ((a_numberOfVertices > 1) || (a_numberOfVertices == 0))
929  {
930  m_color.resize(m_numVertices, color);
931  }
932  else
933  {
934  m_color.push_back(color);
935  }
936  }
937  else

939  m_color.clear();
940  }
941 
942  // update tangent data allocation
943  m_useTangentData = a_useTangentData;
944  if (m_useTangentData)
945  {
946  cVector3d tangent(1.0, 0.0, 0.0);
947  if ((a_numberOfVertices > 1) || (a_numberOfVertices == 0))

966  if ((a_numberOfVertices > 1) || (a_numberOfVertices == 0))
