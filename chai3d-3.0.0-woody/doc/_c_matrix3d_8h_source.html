1687  bool toAxisAngle(cVector3d& a_axis,
1688  double& a_angle) const
1689  {
1690  double angle,x,y,z; // variables for result
1691  double epsilon1 = 0.01; // margin to allow for rounding errors
1692  double epsilon2 = 0.1; // margin to distinguish between 0 and 180 degrees
1693 
1694  if ((abs((*this)(0,1) - (*this)(1,0)) < epsilon1) &&
1695  (abs((*this)(0,2) - (*this)(2,0)) < epsilon1) &&
1696  (abs((*this)(1,2) - (*this)(2,1)) < epsilon1))
1697  {
1698  // singularity found
1699  // first check for identity matrix which must have +1 for all terms
1700  // in leading diagonal and zero in other terms
1701  if ( (abs((*this)(0,1) + (*this)(1,0)) < epsilon2) &&
1702  (abs((*this)(0,2) + (*this)(2,0)) < epsilon2) &&
1703  (abs((*this)(1,2) + (*this)(2,1)) < epsilon2) &&
1704  (abs((*this)(0,0) + (*this)(1,1) + (*this)(2,2)-3) < epsilon2))
1705  {
1706  // this singularity is identity matrix so angle = 0
1707  a_axis.set(1,0,0);
1708  a_angle = 0;
1709  return (true);
1710  }
1711 
1712  // otherwise this singularity is angle = 180
1713  angle = C_PI;
1714  double xx = ((*this)(0,0)+1)/2;
1715  double yy = ((*this)(1,1)+1)/2;
1716  double zz = ((*this)(2,2)+1)/2;
1717  double xy = ((*this)(0,1)+(*this)(1,0))/4;
1718  double xz = ((*this)(0,2)+(*this)(2,0))/4;
1719  double yz = ((*this)(1,2)+(*this)(2,1))/4;
1720 
1721  if ((xx > yy) && (xx > zz))
1722  {
1723  // (*this)(0,0) is the largest diagonal term
1724  if (xx < epsilon1)
1725  {
1726  x = 0;
1727  y = 0.7071;
1728  z = 0.7071;
1729  }
1730  else
1731  {
1732  x = sqrt(xx);
1733  y = xy/x;
1734  z = xz/x;
1735  }
1736  }
1737  else if (yy > zz)
1738  {
1739  // (*this)(1,1) is the largest diagonal term
1740  if (yy< epsilon1)
1741  {
1742  x = 0.7071067811865475;
1743  y = 0.0;
1744  z = 0.7071067811865475;
1745  }
1746  else
1747  {
1748  y = sqrt(yy);
1749  x = xy/y;
1750  z = yz/y;
1751  }
1752  }
1753  else
1754  {
1755  // (*this)(2,2) is the largest diagonal term so base result on this
1756  if (zz < epsilon1)
1757  {
1758  x = 0.7071067811865475;
1759  y = 0.7071067811865475;
1760  z = 0.0;
1761  }
1762  else
1763  {
1764  z = sqrt(zz);
1765  x = xz/z;
1766  y = yz/z;
1767  }
1768  }
1769 
1770  a_axis.set(x,y,z);
1771  a_angle = angle;
1772  return (true);
1773  }
1774 
1775  // as we have reached here there are no singularities so we can handle normally
1776  double s = sqrt(((*this)(2,1) - (*this)(1,2))*((*this)(2,1) - (*this)(1,2)) +
1777  ((*this)(0,2) - (*this)(2,0))*((*this)(0,2) - (*this)(2,0)) +
1778  ((*this)(1,0) - (*this)(0,1))*((*this)(1,0) - (*this)(0,1)));
1779 
1780

1782  // caught by singularity test above, but I've left it in just in case
1783  if (abs(s) < 0.001) s=1;
1784 
1785  angle = acos(( (*this)(0,0) + (*this)(1,1) + (*this)(2,2) - 1)/2);
1786  x = ((*this)(2,1) - (*this)(1,2))/s;
1787  y = ((*this)(0,2) - (*this)(2,0))/s;
1788  z = ((*this)(1,0) - (*this)(0,1))/s;
1789 
1790  a_axis.set(x,y,z);
1791  a_angle = angle;
1792 
1793  return (true);
1794  }
