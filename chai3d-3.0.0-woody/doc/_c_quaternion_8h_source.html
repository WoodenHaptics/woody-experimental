237  inline void toRotMat(cMatrix3d& a_matrix) const
238  {
239  double x2 = 2.0*x*x;
240  double y2 = 2.0*y*y;
241  double z2 = 2.0*z*z;
242  double xy = 2.0*x*y;
243  double wz = 2.0*w*z;
244  double xz = 2.0*x*z;
245  double wy = 2.0*w*y;
246  double yz = 2.0*y*z;
247  double wx = 2.0*w*x;
248 
249  a_matrix(0,0) = 1.0 - y2 - z2;
250  a_matrix(0,1) = xy - wz;
251  a_matrix(0,2) = xz + wy;
252  a_matrix(1,0) = xy + wz;
253  a_matrix(1,1) = 1.0 - x2 - z2;
254  a_matrix(1,2) = yz - wx;
255  a_matrix(2,0) = xz - wy;
256  a_matrix(2,1) = yz + wx;
257  a_matrix(2,2) = 1.0 - x2 - y2;
258  }

273  inline void fromRotMat(cMatrix3d const& a_matrix)
274  {
275  double trace = 1.0 + a_matrix(0,0) + a_matrix(1,1) + a_matrix(2,2);
276 
277  if (trace>0.00000001)
278  {
279  double s = 2.0*sqrt(trace);
280  x = (a_matrix(2,1) - a_matrix(1,2))/s;
281  y = (a_matrix(0,2) - a_matrix(2,0))/s;
282  z = (a_matrix(1,0) - a_matrix(0,1))/s;
283  w = 0.25*s;
284  }
285  else if ((a_matrix(0,0) > a_matrix(1,1)) && (a_matrix(0,0) > a_matrix(2,2)))
286  {
287  // column 1 has largest diagonal
288  double s = 2.0*sqrt(1.0+a_matrix(0,0)-a_matrix(1,1)-a_matrix(2,2));
289  x = 0.25*s;
290  y = (a_matrix(1,0) + a_matrix(0,1))/s;
291  z = (a_matrix(0,2) + a_matrix(2,0))/s;
292  w = (a_matrix(2,1) - a_matrix(1,2))/s;
293  }
294  else if (a_matrix(1,1) > a_matrix(2,2))
295  {
296  // column 2 has largest diagonal
297  double s = 2.0*sqrt(1.0+a_matrix(1,1)-a_matrix(0,0)-a_matrix(2,2));
298  x = (a_matrix(1,0) + a_matrix(0,1))/s;
299  y = 0.25*s;
300  z = (a_matrix(2,1) + a_matrix(1,2))/s;
301  w = (a_matrix(0,2) - a_matrix(2,0))/s;
302  }
303  else
304  {
305  // column 3 has largest diagonal
306  double s = 2.0*sqrt(1.0+a_matrix(2,2)-a_matrix(0,0)-a_matrix(1,1));
307  x = (a_matrix(0,2) + a_matrix(2,0))/s;
308  y = (a_matrix(2,1) + a_matrix(1,2))/s;
309  z = 0.25*s;
310  w = (a_matrix(1,0) - a_matrix(0,1))/s;
311  }
312  }

328  inline void fromAxisAngle(cVector3d a_axis, double a_angleRad)
329  {
330  // not that axis is passed by value so that we can normalize it
331  a_axis.normalize();
332  double sina = sin(a_angleRad / 2.0);

334  w = cosa;
335  x = a_axis(0) * sina;
336  y = a_axis(1) * sina;
337  z = a_axis(2) * sina;

503  inline void slerp(double a_level,
504  cQuaternion const& a_quaternion0,
505  cQuaternion a_quaternion1)
506  {
507  // a_quaternion1 is passed by value so that we can scale it, etc.
508  // compute angle between a_quaternion0 and a_quaternion1
509  double costheta = a_quaternion0.dot(a_quaternion1);
510  if ((costheta-1.0) < 1e-4 && (costheta-1.0) > -1e-4)
511  {
512  // quaternions are parallel
513  // linearly interpolate and normalize
514  *this = a_quaternion0;
515  this->mul(1.0-a_level);
516  a_quaternion1.mul(a_level);
517  this->operator +=(a_quaternion1);
518  this->normalize();
519  }
520  else
521  {
522  double ratio1, ratio2;
523  if ((costheta+1.0) > -1e-4 && (costheta+1.0) < 1e-4)
524  {
525  // a_quaternion0 and a_quaternion1 are 180 degrees apart
526  // there is no unique path between them
527  a_quaternion1.w = a_quaternion0.z;
528  a_quaternion1.x = -a_quaternion0.y;
529  a_quaternion1.y = a_quaternion0.x;
530  a_quaternion1.z = -a_quaternion0.w;
531  ratio1 = sin(C_PI*(0.5-a_level));
532  ratio2 = sin(C_PI*a_level);
533  }
534  else
535  {
536  if (costheta < 0.0)
537  {
538  costheta = -costheta;
539  a_quaternion1.negate();
540  }
541  double theta = acos(costheta);
542  double sintheta = sin(theta);
543 
544  ratio1 = sin(theta*(1.0-a_level))/sintheta;
545  ratio2 = sin(theta*a_level)/sintheta;
546  }
547  *this = a_quaternion0;
548  this->mul(ratio1);
549  a_quaternion1.mul(ratio2);
550  this->operator +=(a_quaternion1);
551  }
552  }
