345  (*curpos == ' ' || *curpos == '\t' || *curpos == '('))

347  curpos++;
348  }
349 
350  // scan data
351  double ax, ay, az;

353 
354  // make sure the conversion worked
355  if (result !=3) return (false);
356 
357  // copy the values we found
