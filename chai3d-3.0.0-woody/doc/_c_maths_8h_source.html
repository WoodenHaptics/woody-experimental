1219 inline void cProjectPointOnPlane(const cVector3d& a_point,
1220  const cVector3d& a_planePoint0,
1221  const cVector3d& a_planePoint1,
1222  const cVector3d& a_planePoint2,

1227  cVector3d v02 = cSub(a_planePoint2, a_planePoint0);
1228  cVector3d point = cSub(a_point, a_planePoint0);
1229

1231  double m00 = v01(0) * v01(0) + v01(1) * v01(1) + v01(2) * v01(2);
1232  double m01 = v01(0) * v02(0) + v01(1) * v02(1) + v01(2) * v02(2);
1233  double m10 = m01;
1234  double m11 = v02(0) * v02(0) + v02(1) * v02(1) + v02(2) * v02(2);
1235  double det = m00 * m11 - m10 * m01;
1236 
1237  // vector
1238  double vm0 = v01(0) * point(0) + v01(1) * point(1) + v01(2) * point(2);

1243  a_v02 = (1.0 / det) * (-m10 * vm0 + m00 * vm1);

1807 inline bool cIntersectionSegmentTriangle(const cVector3d& a_segmentPointA,
1808  const cVector3d& a_segmentPointB,
1809  const cVector3d& a_triangleVertex0,
1810  const cVector3d& a_triangleVertex1,
1811  const cVector3d& a_triangleVertex2,
1812  const bool a_reportFrontSideCollision,
1813  const bool a_reportBackSideCollision,
1814  cVector3d& a_collisionPoint,
1815  cVector3d& a_collisionNormal,
1816  double& a_collisionPosVertex01,
1817  double& a_collisionPosVertex02
1818  )
1819 {
1820  // This value controls how close rays can be to parallel to the triangle
1821  // surface before we discard them
1822  const double C_INTERSECT_EPSILON = 10e-14f;
1823 
1824  // compute a ray and check its length
1825  cVector3d rayDir;
1826  a_segmentPointB.subr(a_segmentPointA, rayDir);
1827  double segmentLengthSquare = rayDir.lengthsq();
1828  if (segmentLengthSquare == 0.0) { return (false); }
1829 
1830  // Compute the triangle's normal
1831  cVector3d t_E0, t_E1, t_N;
1832 
1833  a_triangleVertex1.subr(a_triangleVertex0, t_E0);
1834  a_triangleVertex2.subr(a_triangleVertex0, t_E1);
1835  t_E0.crossr(t_E1, t_N);
1836 
1837  // If the ray is parallel to the triangle (perpendicular to the
1838  // normal), there's no collision
1839  double dotProduct = t_N.dot(rayDir);
1840  if (fabs(dotProduct)<10E-15f) return (false);
1841 
1842  if ((dotProduct < 0.0) && (a_reportFrontSideCollision == false)) return (false);
1843  if ((dotProduct > 0.0) && (a_reportBackSideCollision == false)) return (false);
1844 
1845  double t_T = cDot(t_N, cSub(a_triangleVertex0, a_segmentPointA)) / cDot(t_N, rayDir);
1846 
1847  if (t_T + C_INTERSECT_EPSILON < 0) return (false);
1848 
1849  cVector3d t_Q = cSub(cAdd(a_segmentPointA, cMul(t_T, rayDir)), a_triangleVertex0);
1850  double t_Q0 = cDot(t_E0,t_Q);
1851  double t_Q1 = cDot(t_E1,t_Q);
1852  double t_E00 = cDot(t_E0,t_E0);
1853  double t_E01 = cDot(t_E0,t_E1);
1854  double t_E11 = cDot(t_E1,t_E1);
1855  double t_D = (t_E00 * t_E11) - (t_E01 * t_E01);
1856 
1857  if ((t_D > -C_INTERSECT_EPSILON) && (t_D < C_INTERSECT_EPSILON)) return(false);
1858 
1859  double t_S0 = ((t_E11 * t_Q0) - (t_E01 * t_Q1)) / t_D;
1860  double t_S1 = ((t_E00 * t_Q1) - (t_E01 * t_Q0)) / t_D;
1861 
1862  // Collision has occurred. It is reported.
1863  if (
1864  (t_S0 >= 0.0 - C_INTERSECT_EPSILON) &&
1865  (t_S1 >= 0.0 - C_INTERSECT_EPSILON) &&
1866  ((t_S0 + t_S1) <= 1.0 + C_INTERSECT_EPSILON)
1867  )
1868  {
1869  cVector3d t_I = cAdd(a_triangleVertex0, cMul(t_S0,t_E0), cMul(t_S1, t_E1));
1870 
1871  // Square distance between ray origin and collision point.
1872  double distanceSquare = a_segmentPointA.distancesq(t_I);
1873 
1874  // check if collision occurred within segment. If yes, report collision
1875  if (distanceSquare <= segmentLengthSquare)
1876  {
1877  a_collisionPoint = cAdd(a_segmentPointA, cMul(t_T, rayDir));
1878  t_N.normalizer(a_collisionNormal);
1879  if (cCosAngle(a_collisionNormal, rayDir) > 0.0)
1880  {
1881  a_collisionNormal.negate();
1882  }
1883

1885  a_collisionPosVertex02 = t_S1;
1886

1888  }
1889  }
1890

1892  return (false);
1893 }
