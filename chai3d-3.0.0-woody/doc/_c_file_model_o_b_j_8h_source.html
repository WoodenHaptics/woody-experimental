138 struct ltVertexIndexSet
139 {
140  bool operator()(vertexIndexSet v1, vertexIndexSet v2) const
141  {
142  if (v1.vIndex < v2.vIndex) return 1;
143  if (v2.vIndex < v1.vIndex) return 0;
144  if (v1.nIndex < v2.nIndex) return 1;
145  if (v2.nIndex < v1.nIndex) return 0;
146  if (v1.tIndex < v2.tIndex) return 1;
147  return 0;
148  }
