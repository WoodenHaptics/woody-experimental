232  inline void setFrustumMatrix(double l,
233  double r,
234  double b,
235  double t,
236  double n,
237  double f)
238  {
239  m[0][0] = (2.0*n) / (r-l);
240  m[0][1] = 0.0;
241  m[0][2] = 0.0;
242  m[0][3] = 0.0;
243 
244  m[1][0] = 0.0;
245  m[1][1] = (2.0*n) / (t-b);
246  m[1][2] = 0.0;
247  m[1][3] = 0.0;
248 
249  m[2][0] = (r+l) / (r-l);
250  m[2][1] = (t+b) / (t-b);
251  m[2][2] = -(f+n) / (f-n);
252  m[2][3] = -1.0;
253 
254  m[3][0] = 0.0;
255  m[3][1] = 0.0;
256  m[3][2] = -(2.0*f*n) / (f-n);
257  m[3][3] = 0.0;
258 
259  m_flagTransform = true;
260  }

268  inline void setLookAtMatrix(const double a_eyeX,
269  const double a_eyeY,
270  const double a_eyeZ,
271  const double a_centerX,
272  const double a_centerY,
273  const double a_centerZ,
274  const double a_upX,
275  const double a_upY,
276  const double a_upZ)
277  {
278  double x[3], y[3], z[3];
279  double mag;
280 
281  // create rotation matrix
282 
283  // Z vector
284  z[0] = a_eyeX - a_centerX;
285  z[1] = a_eyeY - a_centerY;
286  z[2] = a_eyeZ - a_centerZ;
287 
288  mag = sqrt( z[0]*z[0] + z[1]*z[1] + z[2]*z[2] );
289  if (mag > 0.0)
290  {
291  z[0] /= mag;
292  z[1] /= mag;
293  z[2] /= mag;
294  }
295 
296  // Y vector
297  y[0] = a_upX;
298  y[1] = a_upY;
299  y[2] = a_upZ;
300 
301  // X vector = Y cross Z
302  x[0] = y[1]*z[2] - y[2]*z[1];
303  x[1] = -y[0]*z[2] + y[2]*z[0];
304  x[2] = y[0]*z[1] - y[1]*z[0];
305 
306  // Recompute Y = Z cross X
307  y[0] = z[1]*x[2] - z[2]*x[1];
308  y[1] = -z[0]*x[2] + z[2]*x[0];
309  y[2] = z[0]*x[1] - z[1]*x[0];
310 
311 
312  // Normalize
313  mag = sqrt( x[0]*x[0] + x[1]*x[1] + x[2]*x[2] );
314  if (mag) {
315  x[0] /= mag;
316  x[1] /= mag;
317  x[2] /= mag;
318  }
319 
320  mag = sqrt( y[0]*y[0] + y[1]*y[1] + y[2]*y[2] );
321  if (mag) {
322  y[0] /= mag;
323  y[1] /= mag;
324  y[2] /= mag;
325  }
326 
327  m[0][0] = x[0]; m[1][0] = x[1]; m[2][0] = x[2]; m[3][0] = -x[0]*a_eyeX + -x[1]*a_eyeY + -x[2]*a_eyeZ;
328  m[0][1] = y[0]; m[1][1] = y[1]; m[2][1] = y[2]; m[3][1] = -y[0]*a_eyeX + -y[1]*a_eyeY + -y[2]*a_eyeZ;
329  m[0][2] = z[0]; m[1][2] = z[1]; m[2][2] = z[2]; m[3][2] = -z[0]*a_eyeX + -z[1]*a_eyeY + -z[2]*a_eyeZ;
330  m[0][3] = 0.0; m[1][3] = 0.0; m[2][3] = 0.0; m[3][3] = 1.0;
331 
332  m_flagTransform = true;
333  }

361  const double a_zFar)
362  {

675  bool inline invert()
676  {
677  if (m_flagTransform)
678  {
679  // transpose 3x3 rotation matrix R
680  double t;
681  t = m[0][1]; m[0][1] = m[1][0]; m[1][0] = t;
682  t = m[0][2]; m[0][2] = m[2][0]; m[2][0] = t;
683  t = m[1][2]; m[1][2] = m[2][1]; m[2][1] = t;
684 
685  // compute -R{t} * p
686  double x = - m[0][0] * m[3][0] - m[1][0] * m[3][1] - m[2][0] * m[3][2];
687  double y = - m[0][1] * m[3][0] - m[1][1] * m[3][1] - m[2][1] * m[3][2];
688  double z = - m[0][2] * m[3][0] - m[1][2] * m[3][1] - m[2][2] * m[3][2];
689 
690  m[3][0] = x;
691  m[3][1] = y;
692  m[3][2] = z;
693 
694  return (true);
695  }
696  else
697  {
698  // Macros used during inversion
699  #ifndef DOXYGEN_SHOULD_SKIP_THIS
700  #define SWAP_ROWS(a, b) { double *_tmp = a; (a)=(b); (b)=_tmp; }
701  #define MAT(m,r,c) (m)[(c)*4+(r)]
702  #endif
703  double *mat = m[0];
704 
705  double wtmp[4][8];
706  double m0, m1, m2, m3, s;
707  double *r0, *r1, *r2, *r3;
708 
709  r0 = wtmp[0], r1 = wtmp[1], r2 = wtmp[2], r3 = wtmp[3];
710 
711  r0[0] = MAT(mat,0,0), r0[1] = MAT(mat,0,1),
712  r0[2] = MAT(mat,0,2), r0[3] = MAT(mat,0,3),
713  r0[4] = 1.0, r0[5] = r0[6] = r0[7] = 0.0,
714 
715  r1[0] = MAT(mat,1,0), r1[1] = MAT(mat,1,1),
716  r1[2] = MAT(mat,1,2), r1[3] = MAT(mat,1,3),
717  r1[5] = 1.0, r1[4] = r1[6] = r1[7] = 0.0,
718 
719  r2[0] = MAT(mat,2,0), r2[1] = MAT(mat,2,1),
720  r2[2] = MAT(mat,2,2), r2[3] = MAT(mat,2,3),
721  r2[6] = 1.0, r2[4] = r2[5] = r2[7] = 0.0,
722 
723  r3[0] = MAT(mat,3,0), r3[1] = MAT(mat,3,1),
724  r3[2] = MAT(mat,3,2), r3[3] = MAT(mat,3,3),
725  r3[7] = 1.0, r3[4] = r3[5] = r3[6] = 0.0;
726 
727  // choose pivot
728  if (fabs(r3[0])>fabs(r2[0])) SWAP_ROWS(r3, r2);
729  if (fabs(r2[0])>fabs(r1[0])) SWAP_ROWS(r2, r1);
730  if (fabs(r1[0])>fabs(r0[0])) SWAP_ROWS(r1, r0);
731  if (0.0 == r0[0])
732  {
733  return (false);
734  }
735 
736  // eliminate first variable
737  m1 = r1[0]/r0[0]; m2 = r2[0]/r0[0]; m3 = r3[0]/r0[0];
738  s = r0[1]; r1[1] -= m1 * s; r2[1] -= m2 * s; r3[1] -= m3 * s;
739  s = r0[2]; r1[2] -= m1 * s; r2[2] -= m2 * s; r3[2] -= m3 * s;
740  s = r0[3]; r1[3] -= m1 * s; r2[3] -= m2 * s; r3[3] -= m3 * s;
741  s = r0[4];
742  if (s != 0.0) { r1[4] -= m1 * s; r2[4] -= m2 * s; r3[4] -= m3 * s; }
743  s = r0[5];
744  if (s != 0.0) { r1[5] -= m1 * s; r2[5] -= m2 * s; r3[5] -= m3 * s; }
745  s = r0[6];
746  if (s != 0.0) { r1[6] -= m1 * s; r2[6] -= m2 * s; r3[6] -= m3 * s; }
747  s = r0[7];
748  if (s != 0.0) { r1[7] -= m1 * s; r2[7] -= m2 * s; r3[7] -= m3 * s; }
749 
750  // choose pivot
751  if (fabs(r3[1])>fabs(r2[1])) SWAP_ROWS(r3, r2);
752  if (fabs(r2[1])>fabs(r1[1])) SWAP_ROWS(r2, r1);
753  if (0.0 == r1[1])
754  {
755  return (false);
756  }
757 
758  // eliminate second variable
759  m2 = r2[1]/r1[1]; m3 = r3[1]/r1[1];
760  r2[2] -= m2 * r1[2]; r3[2] -= m3 * r1[2];
761  r2[3] -= m2 * r1[3]; r3[3] -= m3 * r1[3];
762  s = r1[4]; if (0.0 != s) { r2[4] -= m2 * s; r3[4] -= m3 * s; }
763  s = r1[5]; if (0.0 != s) { r2[5] -= m2 * s; r3[5] -= m3 * s; }
764  s = r1[6]; if (0.0 != s) { r2[6] -= m2 * s; r3[6] -= m3 * s; }
765  s = r1[7]; if (0.0 != s) { r2[7] -= m2 * s; r3[7] -= m3 * s; }
766 
767  // choose pivot
768  if (fabs(r3[2])>fabs(r2[2])) SWAP_ROWS(r3, r2);
769  if (0.0 == r2[2])
770  {
771  return (false);
772  }
773 
774  // eliminate third variable
775  m3 = r3[2]/r2[2];
776  r3[3] -= m3 * r2[3], r3[4] -= m3 * r2[4],
777  r3[5] -= m3 * r2[5], r3[6] -= m3 * r2[6],
778  r3[7] -= m3 * r2[7];
779 
780  // last check
781  if (0.0 == r3[3])
782  {
783  return (false);
784  }
785 
786  s = 1.0/r3[3];
787  r3[4] *= s; r3[5] *= s; r3[6] *= s; r3[7] *= s;
788 
789  m2 = r2[3];
790  s = 1.0/r2[2];
791  r2[4] = s * (r2[4] - r3[4] * m2), r2[5] = s * (r2[5] - r3[5] * m2),
792  r2[6] = s * (r2[6] - r3[6] * m2), r2[7] = s * (r2[7] - r3[7] * m2);
793  m1 = r1[3];
794  r1[4] -= r3[4] * m1, r1[5] -= r3[5] * m1,
795  r1[6] -= r3[6] * m1, r1[7] -= r3[7] * m1;
796  m0 = r0[3];
797  r0[4] -= r3[4] * m0, r0[5] -= r3[5] * m0,
798  r0[6] -= r3[6] * m0, r0[7] -= r3[7] * m0;
799 
800  m1 = r1[2];
801  s = 1.0/r1[1];
802  r1[4] = s * (r1[4] - r2[4] * m1), r1[5] = s * (r1[5] - r2[5] * m1),
803  r1[6] = s * (r1[6] - r2[6] * m1), r1[7] = s * (r1[7] - r2[7] * m1);
804  m0 = r0[2];
805  r0[4] -= r2[4] * m0, r0[5] -= r2[5] * m0,
806  r0[6] -= r2[6] * m0, r0[7] -= r2[7] * m0;
807 
808  m0 = r0[1];
809  s = 1.0/r0[0];
810  r0[4] = s * (r0[4] - r1[4] * m0), r0[5] = s * (r0[5] - r1[5] * m0),
811  r0[6] = s * (r0[6] - r1[6] * m0), r0[7] = s * (r0[7] - r1[7] * m0);
812 
813  MAT(mat,0,0) = r0[4]; MAT(mat,0,1) = r0[5],
814  MAT(mat,0,2) = r0[6]; MAT(mat,0,3) = r0[7],
815  MAT(mat,1,0) = r1[4]; MAT(mat,1,1) = r1[5],
816  MAT(mat,1,2) = r1[6]; MAT(mat,1,3) = r1[7],
817  MAT(mat,2,0) = r2[4]; MAT(mat,2,1) = r2[5],
818  MAT(mat,2,2) = r2[6]; MAT(mat,2,3) = r2[7],
819  MAT(mat,3,0) = r3[4]; MAT(mat,3,1) = r3[5],
820  MAT(mat,3,2) = r3[6]; MAT(mat,3,3) = r3[7];
821 
822  return (true);
823 
824  // Macros used during inversion
825  #undef MAT
826  #undef SWAP_ROWS
827  }
